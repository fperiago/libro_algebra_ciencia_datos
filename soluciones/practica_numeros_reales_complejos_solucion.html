# Gráfica de f(t)

import matplotlib.pyplot as plt

t = np.linspace(0,  4 * np.pi, 1000)
f = 1j * (1 / (1 + t)) * np.exp( 1j * (t + np.pi / 2))

x = np.real(f)
y = np.imag(f)

fig, ax = plt.subplots()
ax.plot(x, y)
ax.set_title('Gráfica de $f(t) = j 1 / (1 + t) e^{ j (t + \pi / 2)}, \quad 0\leq t\leq 4\pi $')

plt.show()

# Gráficas de las parte real e imaginaria de $f$

fig, ax = plt.subplots()
ax.plot(t, x, label='parte real de $f$')
ax.plot(t, y, label='parte imaginaria de $f$')
ax.legend()
